import { useEffect, useState } from 'react';
import { Image, View, FlatList, Text } from 'react-native';
import { getFirestore, doc, getDoc } from "firebase/firestore";

const App = () => {
  const [images, setImages] = useState([]);

  useEffect(() => {
    const fetchImages = async () => {
      const db = getFirestore();
      const docRef = doc(db, 'suaColecao', 'seuDocumento');
      const docSnap = await getDoc(docRef);

      if (docSnap.exists()) {
        setImages(docSnap.data().titulos);
      } else {
        console.log('No such document!');
      }
    };

    fetchImages();
  }, []);

  return (
    <View>
      {images.map((image, index) => (
        <View style={styleCardapio.styleCard}>
          <Image
            source={{ uri: image }}
            style={styleCardapio.styleImage}
          />
          <View>
            <Text style={styleCardapio.pizzaTitle}>Título {index + 1}</Text>
            <Text style={styleCardapio.precoAlign}>Preço {index + 1}</Text>
          </View>
        </View>
      ))}
    </View>
  );
};

export default App;
